# Boxplot
# 
# gnuplot commands to create the boxplot
set terminal pdf monochrome linewidth 1.0 solid size 7.0cm,8.0cm

set border 2 front lt black linewidth 1.000 dashtype solid
set boxwidth 0.5 absolute
set style fill solid 0.5 border lt -1
unset key
set pointsize 0.5
set style data boxplot

set ytics border in scale 0.25,0.25 nomirror norotate autojustify
set ylabel "time in s"

set xtics ("" 1)  scale 0.0
set xtics border in scale 0,0 nomirror norotate autojustify
set xtics norangelimit

#
# generate box plots
#
set title "No Crypto Plugin (500 settings)"
set output "boxplot_500_get_no_crypto_plugin.pdf"
plot '500_get_no_crypto_plugin.dat'

set title "Fcrypt (500 settings)"
set output "boxplot_500_get_fcrypt.pdf"
plot '500_get_fcrypt.dat'

set title "Crypto Plugin (500 settings)"
set xtics ("libgcrypt" 1, "OpenSSL" 2)
set y2tics border in scale 0.25,0.25 nomirror norotate autojustify
set yrange [5.3:5.5]
set y2range [8.15:8.35]
set output "boxplot_500_get_crypto_gcrypt_openssl.pdf"
plot '500_get_crypto_gcrypt.dat' axes x1y1, '500_get_crypto_openssl.dat' axes x2y2
